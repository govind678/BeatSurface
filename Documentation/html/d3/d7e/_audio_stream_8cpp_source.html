74 void AudioStream::audioDeviceIOCallback( const float** inputChannelData,
75  int totalNumInputChannels,
76  float** outputChannelData,
77  int totalNumOutputChannels,
78  int blockSize)
79 {
80 
81  if (BeatSurfaceBase::eCurrentMode == BeatSurfaceBase::CustomTrainMode)
82  {
83  if(m_pcOnsetClassifier->detectOnsets(inputChannelData))
84  {
85  m_pcOnsetClassifier->train(m_iCurrentClassIndexToTrain);
86  guiUpdater->displayTrainingOnset(m_iCurrentClassIndexToTrain);
87 
88  m_ppfOnsetAudio.add(new AudioSampleBuffer(1, blockSize));
89 
90  for (int sample = 0; sample < blockSize; sample++)
91  {
92  m_ppfOnsetAudio.getLast()->copyFrom(0, 0, inputChannelData[0], blockSize);
93  }
94 
95  m_piClassLabels.add(m_iCurrentClassIndexToTrain);
96 
97  guiUpdater->drawSpectralCentroid(m_pcOnsetClassifier->getCurrentSpectralCentroid());
98  guiUpdater->drawSpectrum(m_pcOnsetClassifier->getCurrentSpectrum());
99  }
100  }
101 
102 
103  else if (BeatSurfaceBase::eCurrentMode == BeatSurfaceBase::AudioTrainMode)
104  {
105  audioSourcePlayer.audioDeviceIOCallback (inputChannelData,
106  totalNumInputChannels,
107  outputChannelData,
108  totalNumOutputChannels,
109  blockSize);
110  }
111 
112 
113 
114  else if (BeatSurfaceBase::eCurrentMode == BeatSurfaceBase::PlayMode)
115  {
116  m_fCurrentRMS = m_pcOnsetClassifier->rootMeanSquare(inputChannelData);
117  guiUpdater->drawWaveformWithSample(m_fCurrentRMS);
118 
119 
120  if (m_pcOnsetClassifier->detectOnsets(inputChannelData))
121  {
122  if(m_pcOnsetClassifier->getNumClasses() > 0)
123  {
124  m_iCurrentClassificationResult = m_pcOnsetClassifier->classify();
125  guiUpdater->displayPlayingOnset(m_iCurrentClassificationResult);
126  m_pcMidiOut->makeNoteAndSend(1, m_iCurrentClassificationResult + 35, int((m_fCurrentRMS * 120.0f) / 0.6f), 250);
127  }
128 
129  guiUpdater->drawSpectralCentroid(m_pcOnsetClassifier->getCurrentSpectralCentroid());
130  guiUpdater->drawSpectrum(m_pcOnsetClassifier->getCurrentSpectrum());
131  }
132  }
133 
134 
135  else if (BeatSurfaceBase::eCurrentMode == BeatSurfaceBase::IdleMode)
136  {
137 // guiUpdater->drawWaveformWithSample(m_pcOnsetClassifier->rootMeanSquare(inputChannelData));
138 
139  for (int channel = 0; channel < totalNumOutputChannels; channel++)
140  {
141  if (m_ppfOnsetAudio.size() > 0)
142  {
143  for (int sample = 0; sample < blockSize; sample++)
144  {
145  outputChannelData[channel][sample] = m_ppfOnsetAudio.getUnchecked(m_iCurrentIndexToPlay)->getSampleData(0)[sample];
146  }
147  }
148  }
149 
150 
151  }
