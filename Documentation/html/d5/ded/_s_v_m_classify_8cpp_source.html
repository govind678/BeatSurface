36 double* SVMClassify::classify(float *pfFeatures, int iNumFeatures)
37 {
38 
39  if (!m_bIsInitialized)
40  return 0;
41 
42  int iNodeIdx = 0;
43  svm_node *pSvmFeatures = new svm_node [iNumFeatures + 1];
44 
45  // copy feature data to svm array
46  for (int i = 0; i < iNumFeatures; i++)
47  {
48  // libsvm expects "sparse" vectors e.g. elements with a value of 0 are left out
49  if (pfFeatures[i] == 0)
50  continue;
51  pSvmFeatures[iNodeIdx].index = i;
52  pSvmFeatures[iNodeIdx].value = pfFeatures[i];
53  iNodeIdx++;
54  }
55  pSvmFeatures[iNodeIdx].index = -1;
56 
57 
58  m_dResult = svm_predict_probability (m_pSVMModel, pSvmFeatures, m_pdProbability);
59 
60  delete [] pSvmFeatures;
61 
62  return m_pdProbability;
63 }
